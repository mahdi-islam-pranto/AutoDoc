-- create doctor table
CREATE TABLE IF NOT EXISTS doctors (
    id                      SERIAL PRIMARY KEY,
    name                    VARCHAR(255)        NOT NULL,
    doctor_no               VARCHAR(50)         UNIQUE,
    degree                  VARCHAR(255),
    designation             VARCHAR(100),
    gender                  VARCHAR(20),
    registration_no         VARCHAR(50),
    experience_years        INTEGER,
    first_visit_fee         NUMERIC(10, 2),
    follow_up_fee           NUMERIC(10, 2),
    follow_up_validity_days INTEGER,
    location                TEXT,
    chamber_address         VARCHAR(100),
    avg_duration            INTEGER,
    avg_load                INTEGER,
    reserved                INTEGER,
    email                   VARCHAR(255),
    phone                   VARCHAR(100),
    priority                VARCHAR(50),
    priority_serial         INTEGER,
    description             TEXT,
    opening_hours           JSONB,
    is_active               BOOLEAN             DEFAULT TRUE,
    branch_name             VARCHAR(255),
    specialization_name     VARCHAR(100),
    created_at              TIMESTAMP,
    updated_at              TIMESTAMP
);

-- create appointments table

CREATE TABLE IF NOT EXISTS appointments (
    id                      SERIAL PRIMARY KEY,

    -- Doctor reference
    doctor_id               INTEGER             NOT NULL REFERENCES doctors(id) ON DELETE RESTRICT,

    -- Patient info (no patients table assumed, stored inline)
    patient_name            VARCHAR(255)        NOT NULL,
    patient_phone           VARCHAR(50)         NOT NULL,
    patient_email           VARCHAR(255),
    patient_age             INTEGER,
    patient_gender          VARCHAR(20),

    -- Appointment details
    appointment_date        DATE                NOT NULL,
    appointment_slot        TIME                NOT NULL,  -- e.g. 16:00
    visit_type              VARCHAR(20)         NOT NULL DEFAULT 'first_visit'
                                CHECK (visit_type IN ('first_visit', 'follow_up')),

    -- Fee snapshot (copied from doctors at booking time, not joined live)
    -- fee                     NUMERIC(10, 2)      NOT NULL,

    -- Status tracking
    status                  VARCHAR(20)         NOT NULL DEFAULT 'confirmed'
                                CHECK (status IN ('confirmed', 'cancelled', 'completed', 'no_show')),

    -- Serial number for the day (e.g. patient is 3rd in queue)
    serial_no               INTEGER,

    -- Source of booking
    booked_via              VARCHAR(50)         DEFAULT 'ai_agent'
                                CHECK (booked_via IN ('ai_agent', 'web', 'phone', 'walk_in')),

    notes                   TEXT,

    created_at              TIMESTAMP           DEFAULT NOW(),
    updated_at              TIMESTAMP           DEFAULT NOW()
);
